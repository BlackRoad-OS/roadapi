Quick Start

# Install the SDK npm install @blackroad/roadapi # Or use cURL curl -X GET "https://api.blackroad.io/v1/agents" \ -H "Authorization: Bearer YOUR_API_KEY" \ -H "Content-Type: application/json" # JavaScript Example import { RoadAPI } from '@blackroad/roadapi'; const api = new RoadAPI({ apiKey: 'sk_live_your_key_here' }); // List all AI agents const agents = await api.agents.list(); console.log(`Found ${agents.length} agents`); // Deploy a new agent const newAgent = await api.agents.create({ name: 'my-agent', model: 'claude-sonnet-4', cluster: 'pi-cluster-1' }); // Run AI inference const result = await api.ai.inference({ prompt: 'Hello, AI!', max_tokens: 100 }); console.log(result.text);
